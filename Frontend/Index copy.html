async function markWorkStart() { const selectedName = document.getElementById("employeeName").value; const confirmName = document.getElementById("confirmName").value; if (selectedName !== confirmName) { showErrorMessage("taskErrorMessage", "Names don't match!"); return; } try { const response = await fetch('http://localhost:5000/api/employees/start-work', { method: 'POST', headers: { 'Content-Type': 'application/json' }, body: JSON.stringify({ name: selectedName, confirmName }) }); const data = await response.json(); if (response.ok) { showStatusMessage("taskStatusMessage", data.message); } else { showErrorMessage("taskErrorMessage", data.error); } } catch (error) { showErrorMessage("taskErrorMessage", "Failed to connect to server"); } }
